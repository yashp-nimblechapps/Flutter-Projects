9) StatefulWidget and setState and made a counter app
(counter increase when button clicked)

class MyHomePage extends StatefulWidget {
  const MyHomePage({super.key, required this.title});

  final String title;

  @override
  State<MyHomePage> createState() => MyHomeState();
}

class MyHomeState extends State<MyHomePage> {

  var count = 0;

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('Stateful'),
      ),
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            Text('Count: $count', style: TextStyle(fontSize: 34),),
            ElevatedButton(
              onPressed: () {
                setState(() {
                  count++;
                 print("Count: $count");
                });

              }, 
              child: Text('Increment Counter')
            )
          ],

        ),
      ),

    );
  }
}

10) StatefulWidget and setState and made a basic calculator app
(add, sub, mul, divide)

class MyHomeState extends State<MyHomePage> {

  var no1 = TextEditingController();
  var no2 = TextEditingController();
  var result;


  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('Basic calculator'),
      ),
      body: Container(
        color: Colors.lightBlueAccent,

        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            TextField(
              controller: no1,
              keyboardType: TextInputType.number,
            ),

            SizedBox(height: 20,),

            TextField(
              controller: no2,
              keyboardType: TextInputType.number,

            ),

            SizedBox(height: 20,),

            Row(
              mainAxisAlignment: MainAxisAlignment.spaceAround,
              children: [
                ElevatedButton(
                  onPressed: () {                    
                      int num1 = int.parse(no1.text);
                      int num2 = int.parse(no2.text);
                      var sum  = num1 + num2;

                      result = "Sum is $sum";
                      setState(() {});
                  }, 
                  child: Text('ADD')
                ),
                ElevatedButton(
                  onPressed: () {
                      int num1 = int.parse(no1.text);
                      int num2 = int.parse(no2.text);
                      var sub = num1 - num2;

                      result = "Sub is $sub";                  
                      setState(() {});
                  }, 
                  child: Text('SUB')
                ),
                ElevatedButton(
                  onPressed: () {
                      int num1 = int.parse(no1.text);
                      int num2 = int.parse(no2.text);
                      var mul = num1 * num2;

                      result = "Mul is $mul";
                      setState(() {});
                  }, 
                  child: Text('MUL')
                ),
                ElevatedButton(
                  onPressed: () {
                      int num1 = int.parse(no1.text);
                      int num2 = int.parse(no2.text);
                      var divide = num1 / num2;

                      result = "Div is ${divide.toStringAsFixed(2)}";
                      setState(() {});
                  }, 
                  child: Text('DIVIDE')
                ),
              ],              
            ),
            SizedBox(height: 20,),

            Text('$result', style: TextStyle(fontSize: 34),),
          ],

        ),
      ),

    );
  }
}

11) List Item mapped directly to ListView
Input is List (an array)

class MyHomeState extends State<MyHomePage> {

  var arrData = [
    'ABC', 'DEF', 'GHI','JKL', 'MNO', 'PQR', 'STU', 'VWX', 'YZ'
  ];

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('MY APP')
      ),
      body: Container(
        child: ListView(
          children: arrData.map((value) {
            return Padding(
              padding: const EdgeInsets.all(8.0),
              child: Container(
                decoration: BoxDecoration(
                  borderRadius: BorderRadius.circular(21),
                  color: Colors.lightBlueAccent,
                ),
                child: Padding(
                  padding: const EdgeInsets.all(8.0),
                  child: Center(child: Text(value)),
                ),
              ),
            );
          }).toList()
        ),
      )
